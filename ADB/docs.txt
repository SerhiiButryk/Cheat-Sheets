--- device rotation ---

--- Command to get Android device info via adb ---

adb shell getprop | \
grep "model\|version.sdk\|manufacturer\|ro.serialno\|product.name\|brand\|version.release\|build.id\|security_patch" | \
sed 's/ro\.//g' && \
echo "[device.imei]: [$(adb shell service call iphonesubinfo 1 | \
awk -F "'" '{print $2}' | \
sed '1 d'| tr -d '\n' | tr -d '.' | tr -d ' ')]"
